/* Define layer order */
@layer background, components, utilities;
/* Background layer - lowest priority */
@layer background {
    .demo-card {
        background: #e3f2fd;
        border: 2px solid #90caf9;
        padding: 2rem;
        border-radius: 12px;
    }
}
/* Components layer - higher priority */
@layer components {
    .layer-button {
        background: #4caf50;
        color: white;
        border: none;
        padding: 1rem 2rem;
        border-radius: 8px;
        cursor: pointer;
    }
}
/* Utilities layer - highest priority */
@layer utilities {
    .highlight {
        background: #ff5722;
        transform: scale(1.05);
    }
}
