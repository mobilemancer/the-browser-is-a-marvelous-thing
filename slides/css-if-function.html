.conditional-card {
    background: if(
        style(data-theme="dark"):
            linear-gradient(135deg, #1f2937, #374151);
        style(data-theme="light"):
            linear-gradient(135deg, #f8fafc, #e2e8f0);
        else:
            linear-gradient(135deg, #8b5cf6, #06b6d4);
    );

    color: if(
        style(data-theme="dark"): #f1f5f9;
        style(data-theme="light"): #1e293b;
        else: white;
    );

    border-color: if(
        style(data-theme="dark"): #8b5cf6;
        style(data-theme="light"): #06b6d4;
        else: #fbbf24;
    );
}
