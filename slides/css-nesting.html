/* Native CSS Nesting */
.card {
    background: white;
    border-radius: 8px;
    padding: 1rem;
    box-shadow: 0 2px 8px rgba(0,0,0,0.1);

    /* Nested selectors */
    & .title {
        font-size: 1.5rem;
        font-weight: bold;
        margin-bottom: 0.5rem;
    }

    & .content {
        color: #666;
        line-height: 1.6;
    }

    /* Pseudo-selectors */
    &:hover {
        transform: translateY(-2px) scale(1.5);
        box-shadow: 0 4px 16px rgba(0,0,0,0.15);
        transition: all 0.3s ease;
    }

    /* Media queries inside selectors */
    @media (max-width: 768px) {
        padding: 0.5rem;
        
        & .title {
            font-size: 1.2rem;
        }
    }
}
