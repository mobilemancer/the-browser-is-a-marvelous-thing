/* Define the anchor */
.anchor-button {
    anchor-name: --my-anchor;
}

/* Position element relative to anchor */
.tooltip {
    position: absolute;
    position-anchor: --my-anchor;

    /* Position at bottom center */
    top: anchor(bottom);
    left: anchor(center);
    translate: -50% 0.5rem;
}

/* Auto-flip when hitting viewport edge */
.tooltip {
    position-try-fallbacks: 
        flip-block, 
        flip-inline;
}
